\begin{frame}{Concept}
\begin{itemize}
    \item For each lane, evaluate safety using:
    \begin{itemize}
        \item front obstacle
        \item rear obstacle
    \end{itemize}
    \item Compute:
    \begin{itemize}
        \item relative distance
        \item relative velocity
        \item Time-to-Collision (TTC)
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Variable Definition}
Let:
\[
d_{\mathrm{rel}} = \text{relative longitudinal distance}
\qquad
v_{\mathrm{rel}} = \text{relative closing speed}
\]

\vspace{0.4em}
For front vehicle:
\[
d_{\mathrm{front}} = x_{\mathrm{obj}} - x_{\mathrm{ego}}
\qquad
v_{\mathrm{close,front}} = v_{\mathrm{ego}} - v_{\mathrm{obj}}
\]

For rear vehicle:
\[
d_{\mathrm{rear}} = x_{\mathrm{ego}} - x_{\mathrm{obj}}
\qquad
v_{\mathrm{close,rear}} = v_{\mathrm{obj}} - v_{\mathrm{ego}}
\]
\end{frame}

\begin{frame}{TTC Formula}
For closing condition:
\[
v_{\mathrm{close}} > 0
\]

\[
\mathrm{TTC} = \frac{d_{\mathrm{rel}}}{v_{\mathrm{close}}}
\]

Else:
\[
\mathrm{TTC} = \infty
\]
\end{frame}

\begin{frame}{Temporal Sampling}
\begin{itemize}
    \item Maintain last \(N\) TTC samples
    \item Typical choice: \(N = 5\text{--}6\)
    \item Fit a line to the TTC samples
\end{itemize}

\[
\mathrm{TTC}_{\mathrm{history}}
=
[\mathrm{TTC}_{k-N+1}, \ldots, \mathrm{TTC}_k]
\]

\vspace{0.4em}
Line fit:
\[
\mathrm{TTC}(i) \approx m i + b
\]

Slope used for trend analysis:
\[
\mathrm{slope} = m
\]
\end{frame}

\begin{frame}{Decision Rule}
For each lane:

\[
\text{if }
\bigl(\mathrm{slope} < 0\bigr)
\;\land\;
\bigl(\mathrm{TTC} < \mathrm{TTC}_{\mathrm{threshold}}\bigr)
\Rightarrow
\text{lane = NOT SAFE}
\]

\[
\text{else if }
\bigl(\mathrm{slope} > 0\bigr)
\Rightarrow
\text{lane = SAFE}
\]

\begin{itemize}
    \item Apply rule to both front and rear obstacle checks
    \item Lane is selected only if safety condition is satisfied
\end{itemize}
\end{frame}

\begin{frame}{Summary}
\begin{itemize}
    \item TTC captures collision risk
    \item Trend (slope) captures risk evolution
    \item Negative slope $\rightarrow$ increasing risk
    \item Positive slope $\rightarrow$ decreasing risk
    \item Used for lane selection in behavior planner
\end{itemize}
\end{frame}
