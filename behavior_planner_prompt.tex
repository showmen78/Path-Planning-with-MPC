\section{Purpose}

This document provides a structured prompt template for a high-level autonomous
vehicle Behavior Planner, together with worked examples. The planner is assumed
to read ego state, HD map context, route information, surrounding vehicle
predictions, and V2X broadcasts, then return a strict JSON response containing
the selected behavior, target speed, outbound CAV broadcast, and reasoning.

\section{Prompt Template}

\subsection{System Instruction}

\begin{quote}
You are the high-level Behavior Planner for an autonomous vehicle (Ego).
Your job is to read the current state of the environment, HD Map, Global Route,
predicted intentions, and V2X broadcasts, then output a strict JSON response.
Safety and ``emergency'' broadcasts supersede all other routing goals.
\end{quote}

\subsection{Allowed Behaviors}

\begin{lstlisting}[style=jsonstyle]
["LANE_KEEP", "LANE_CHANGE_LEFT", "LANE_CHANGE_RIGHT",
 "ACCELERATE", "DECELERATE", "STOP_AT_LINE", "EMERGENCY_BRAKE"]
\end{lstlisting}

\subsection{Input Structure}

\begin{enumerate}
    \item \textbf{Ego Vehicle State}
    \begin{itemize}
        \item State $[x, y, v, \psi]$: \texttt{\{ego\_x\}}, \texttt{\{ego\_y\}}, \texttt{\{ego\_v\}} m/s, \texttt{\{ego\_psi\}} rad
        \item Current Road ID: \texttt{\{ego\_road\_id\}}
        \item Lane ID: \texttt{\{ego\_lane\_id\}}
    \end{itemize}

    \item \textbf{Global Route Plan (A* Navigation)}
    \begin{itemize}
        \item Distance to Destination: \texttt{\{dist\_to\_dest\}} km
        \item Next Macro Maneuver: \texttt{\{a\_star\_next\_maneuver\}}
        \item Optimal Lane for Route: \texttt{\{a\_star\_optimal\_lane\}}
    \end{itemize}

    \item \textbf{CARLA HD Map Context}
    \begin{itemize}
        \item Speed Limit: \texttt{\{speed\_limit\}} m/s
        \item Legal Lane Changes:
        \begin{itemize}
            \item Left: \texttt{\{can\_change\_left\}}
            \item Right: \texttt{\{can\_change\_right\}}
        \end{itemize}
        \item Next Traffic Light State: \texttt{\{traffic\_light\_state\}}
        \item Distance to Traffic Light: \texttt{\{dist\_to\_light\}} m
    \end{itemize}

    \item \textbf{Surrounding Vehicles and Predicted Intentions}
    \begin{itemize}
        \item Vehicle ID: \texttt{\{v1\_id\}}
        \begin{itemize}
            \item State $[x, y, v, \psi]$: \texttt{\{v1\_x\}}, \texttt{\{v1\_y\}}, \texttt{\{v1\_v\}} m/s, \texttt{\{v1\_psi\}} rad
            \item Potential Field Risk: \texttt{\{v1\_risk\_level\}}
            \item Predicted Intention: text description
        \end{itemize}
        \item Vehicle ID: \texttt{\{v2\_id\}}
        \begin{itemize}
            \item State $[x, y, v, \psi]$: \texttt{\{v2\_x\}}, \texttt{\{v2\_y\}}, \texttt{\{v2\_v\}} m/s, \texttt{\{v2\_psi\}} rad
            \item Potential Field Risk: \texttt{\{v2\_risk\_level\}}
            \item Predicted Intention: text description
        \end{itemize}
    \end{itemize}

    \item \textbf{V2X / CAV Broadcasts (Received)}
    \begin{itemize}
        \item Structured list of received dictionaries in the form:
        \begin{lstlisting}[style=jsonstyle]
[
  { "id": "...", "type": "...", "message": "..." }
]
        \end{lstlisting}
    \end{itemize}
\end{enumerate}

\subsection{Task Statement}

\begin{quote}
Determine the safest allowed behavior. Provide a target speed
(\texttt{target\_v}, in m/s), the chosen behavior, a message dictionary to
broadcast to other CAVs, and a brief reasoning.
\end{quote}

\subsection{CAV Broadcast Structure}

The outbound CAV broadcast must always use the exact structure:

\begin{lstlisting}[style=jsonstyle]
{ "id": "...", "type": "...", "message": "..." }
\end{lstlisting}

\noindent with:
\begin{itemize}
    \item \texttt{id}: sender vehicle identifier
    \item \texttt{type}: broadcast category, typically \texttt{normal} or \texttt{emergency}
    \item \texttt{message}: human-readable broadcast message
\end{itemize}

\subsection{Expected JSON Format}

\begin{lstlisting}[style=jsonstyle]
{
  "behavior": "BEHAVIOR_NAME",
  "target_v": 0.0,
  "cav_broadcast": {
    "id": "vehicle_id",
    "type": "normal/emergency",
    "message": "Your broadcast message here"
  },
  "reasoning": "Explanation here"
}
\end{lstlisting}

\section{Example 1: Lane Closure}

\subsection{Prompt Instance}

\begin{enumerate}
    \item \textbf{Ego Vehicle State}
    \begin{itemize}
        \item State $[x, y, v, \psi]$: $100.0$, $10.5$, $15.0$ m/s, $0.0$ rad
        \item Current Road ID: $12$
        \item Lane ID: $3$ 
    \end{itemize}

    \item \textbf{Global Route Plan}
    \begin{itemize}
        \item Distance to Destination: $5.2$ km
        \item Next Macro Maneuver: Continue Straight
        \item Optimal Lane for Route: $3$
    \end{itemize}

    \item \textbf{CARLA HD Map Context}
    \begin{itemize}
        \item Speed Limit: $20.0$ m/s
        \item Legal Lane Changes: Left = True, Right = False
        \item Next Traffic Light State: Green (Distance: $400.0$ m)
        \item Map Warning: Static blockage in Lane 3 at distance $100.0$ m
    \end{itemize}

    \item \textbf{Surrounding Vehicles and Predicted Intentions}
    \begin{itemize}
        \item Vehicle ID: $101$
        \begin{itemize}
            \item State $[x, y, v, \psi]$: $102.0$, $7.0$, $15.0$ m/s, $0.0$ rad
            \item Potential Field Risk: Warning Zone
            \item Predicted Intention: Maintaining lane 2, steady speed
        \end{itemize}
    \end{itemize}

    \item \textbf{V2X / CAV Broadcasts}
    \begin{lstlisting}[style=jsonstyle]
[
  {
    "id": 101,
    "type": "normal",
    "message": "Maintaining speed at 15 m/s."
  }
]
    \end{lstlisting}
\end{enumerate}

\subsection{Gemini Response}

\begin{lstlisting}[style=jsonstyle]
{
  "behavior": "DECELERATE",
  "target_v": 10.0,
  "cav_broadcast": {
    "id": "Ego",
    "type": "normal",
    "message": "Decelerating to yield to Vehicle 101. Preparing to merge into Lane 2 to avoid obstacle."
  },
  "reasoning": "There is a static blockage 100 meters ahead in the current lane (Lane 3). Ego must transition to Lane 2, but Vehicle 101 is currently occupying the adjacent space in the Warning Zone at the same speed (15 m/s). Ego must decelerate to fall behind Vehicle 101, opening up a safe gap to execute a LANE_CHANGE_LEFT in the near future."
}
\end{lstlisting}

\section{Example 2: Adjacent Vehicle Intends to Merge Into Ego Lane}

\subsection{Prompt Instance}

\begin{enumerate}
    \item \textbf{Ego Vehicle State}
    \begin{itemize}
        \item State $[x, y, v, \psi]$: $250.0$, $0.0$, $20.0$ m/s, $0.0$ rad
        \item Current Road ID: $15$
        \item Lane ID: $0$
    \end{itemize}

    \item \textbf{Global Route Plan}
    \begin{itemize}
        \item Distance to Destination: $8.5$ km
        \item Next Macro Maneuver: Continue Straight
        \item Optimal Lane for Route: $0$
    \end{itemize}

    \item \textbf{CARLA HD Map Context}
    \begin{itemize}
        \item Speed Limit: $25.0$ m/s
        \item Legal Lane Changes: Left = False, Right = True
        \item Next Traffic Light State: None
    \end{itemize}

    \item \textbf{Surrounding Vehicles and Predicted Intentions}
    \begin{itemize}
        \item Vehicle ID: $205$ (slightly ahead in right lane 1)
        \begin{itemize}
            \item State $[x, y, v, \psi]$: $265.0$, $3.5$, $18.0$ m/s, $0.1$ rad
            \item Potential Field Risk: Warning Zone
            \item Predicted Intention: Speeding up, will change lane from 1 to 0
        \end{itemize}
    \end{itemize}

    \item \textbf{V2X / CAV Broadcasts}
    \begin{lstlisting}[style=jsonstyle]
[
  {
    "id": 205,
    "type": "normal",
    "message": "Initiating lane change to Lane 0 to pass slower traffic."
  }
]
    \end{lstlisting}
\end{enumerate}

\subsection{Gemini Response}

\begin{lstlisting}[style=jsonstyle]
{
  "behavior": "DECELERATE",
  "target_v": 15.0,
  "cav_broadcast": {
    "id": "Ego",
    "type": "normal",
    "message": "Acknowledged Vehicle 205 merge intent. Decelerating to open a safe gap in Lane 0."
  },
  "reasoning": "Vehicle 205 is 15 meters ahead in Lane 1, has entered the potential field Warning Zone, and broadcasted its intent to cut into Lane 0. Since Ego is traveling faster (20 m/s) than Vehicle 205 (18 m/s), maintaining speed would risk a collision. Ego will decelerate to yield the space, allowing Vehicle 205 to safely merge while Ego remains in its A* optimal lane."
}
\end{lstlisting}

\section{Example 3: Emergency Vehicle Override}

\subsection{Prompt Instance}

\begin{enumerate}
    \item \textbf{Ego Vehicle State}
    \begin{itemize}
        \item State $[x, y, v, \psi]$: $500.0$, $0.0$, $18.0$ m/s, $0.0$ rad
        \item Current Road ID: $22$
        \item Lane ID: $0$ (left-most lane)
    \end{itemize}

    \item \textbf{Global Route Plan}
    \begin{itemize}
        \item Distance to Destination: $3.0$ km
        \item Next Macro Maneuver: Left turn at next intersection
        \item Optimal Lane for Route: $0$
    \end{itemize}

    \item \textbf{CARLA HD Map Context}
    \begin{itemize}
        \item Speed Limit: $20.0$ m/s
        \item Legal Lane Changes: Left = False, Right = True
        \item Next Traffic Light State: Green (Distance: $800.0$ m)
    \end{itemize}

    \item \textbf{Surrounding Vehicles and Predicted Intentions}
    \begin{itemize}
        \item Vehicle ID: $301$ 
        \begin{itemize}
            \item State $[x, y, v, \psi]$: $530.0$, $0.0$, $18.0$ m/s, $0.0$ rad
            \item Potential Field Risk: Safe Zone
            \item Predicted Intention: Maintaining lane 0, steady speed
        \end{itemize}
        \item Vehicle ID: $911$ (ambulance, far behind in ego lane 0)
        \begin{itemize}
            \item State $[x, y, v, \psi]$: $300.0$, $0.0$, $35.0$ m/s, $0.0$ rad
            \item Potential Field Risk: Safe Zone, but closing rapidly
            \item Predicted Intention: High speed, maintaining lane 0
        \end{itemize}
    \end{itemize}

    \item \textbf{V2X / CAV Broadcasts}
    \begin{lstlisting}[style=jsonstyle]
[
  {
    "id": 301,
    "type": "normal",
    "message": "Cruising at 18 m/s."
  },
  {
    "id": 911,
    "type": "emergency",
    "message": "Ambulance responding to code 3. Clear Lane 0 immediately!"
  }
]
    \end{lstlisting}
\end{enumerate}

\subsection{Gemini Response}

\begin{lstlisting}[style=jsonstyle]
{
  "behavior": "LANE_CHANGE_RIGHT",
  "target_v": 18.0,
  "cav_broadcast": {
    "id": "Ego",
    "type": "normal",
    "message": "Yielding to emergency vehicle 911. Executing lane change to Lane 1."
  },
  "reasoning": "Received an emergency broadcast from Vehicle 911 (Ambulance) approaching rapidly from behind in Lane 0. Emergency protocols supersede the A* global route preference for Lane 0. Lane 1 is clear, so executing an immediate right lane change to yield."
}
\end{lstlisting}
